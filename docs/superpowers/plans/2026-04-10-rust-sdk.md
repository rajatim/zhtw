# Rust SDK Implementation Plan

> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** Build a Rust native crate (`zhtw`) + WebAssembly bindings (`zhtw-wasm`) as the 4th SDK for the zhtw monorepo, with full golden-test parity, mono-versioning integration, and CI pipeline.

**Architecture:** Cargo workspace at `sdk/rust/` with two members: `zhtw/` (native crate, `build.rs` pre-compiles daachorse automaton + phf char map) and `zhtw-wasm/` (thin wasm-bindgen wrapper published as npm package). All data comes from the shared `sdk/data/zhtw-data.json` via a crate-local copy synced by `make export`.

**Tech Stack:** Rust 2021 (MSRV 1.80), daachorse =1.0.0, phf 0.11, thiserror 2, serde/serde_json 1, wasm-bindgen 0.2, criterion 0.5, wasm-pack (bundler target)

use crate::Error;

/// Source dictionary selection for conversion.
#[derive(Debug, Clone, Copy, PartialEq, Eq, Hash)]
pub enum Source {
    /// Simplified Chinese (China) → Taiwan Traditional
    Cn,
    /// Hong Kong Traditional → Taiwan Traditional
    Hk,
}

impl std::str::FromStr for Source {
    type Err = Error;
    fn from_str(s: &str) -> std::result::Result<Self, Self::Err> {
        match s {
            "cn" => Ok(Self::Cn),
            "hk" => Ok(Self::Hk),
            _ => Err(Error::InvalidSource(s.to_string())),
        }
    }
}

impl std::fmt::Display for Source {
    fn fmt(&self, f: &mut std::fmt::Formatter<'_>) -> std::fmt::Result {
        match self {
            Self::Cn => write!(f, "cn"),
            Self::Hk => write!(f, "hk"),
        }
    }
}
```

use crate::{Config, Error, Result, Source};

/// Builder for constructing a [`Converter`](crate::Converter) with custom settings.
#[derive(Debug)]
pub struct Builder {
    config: Config,
}

impl Default for Builder {
    fn default() -> Self {
        Self {
            config: Config::default(),
        }
    }
}

impl Builder {
    pub fn sources<I: IntoIterator<Item = Source>>(mut self, sources: I) -> Self {
        self.config.sources = sources.into_iter().collect();
        self
    }

    pub fn custom_dict<I, K, V>(mut self, dict: I) -> Self
    where
        I: IntoIterator<Item = (K, V)>,
        K: Into<String>,
        V: Into<String>,
    {
        self.config.custom_dict = dict
            .into_iter()
            .map(|(k, v)| (k.into(), v.into()))
            .collect();
        self
    }

    /// Validates config and constructs the Converter.
    ///
    /// Returns `Err(EmptySources)` if sources is empty.
    pub fn build(self) -> Result<crate::Converter> {
        crate::Converter::new(self.config)
    }
}
```

```rust
/// Magic header for precompiled automaton blobs.
///
/// Format (28 bytes, all little-endian):
///   [0..8]   magic          = b"ZHTWDAAC"
///   [8..10]  header_version : u16 (currently 1)
///   [10..14] daachorse_ver  : u32 packed as (major<<16 | minor<<8 | patch)
///   [14..22] dict_hash      : [u8; 8]  (first 8 bytes of blake3)
///   [22]     source_mask    : u8  (bit 0 = Cn, bit 1 = Hk)
///   [23..28] reserved       : [u8; 5]  (MBZ)

pub(crate) const ZHTW_AUTOMATON_MAGIC: [u8; 8] = *b"ZHTWDAAC";
pub(crate) const CURRENT_HEADER_VERSION: u16 = 1;
pub(crate) const HEADER_LEN: usize = 28;
pub(crate) const DAACHORSE_VERSION_PACKED: u32 = 0x0001_0000; // 1.0.0

#[derive(Clone, Copy, PartialEq, Eq, Debug)]
pub(crate) struct SourceMask(pub u8);

impl SourceMask {
    pub const CN: Self = SourceMask(0b01);
    pub const HK: Self = SourceMask(0b10);
    pub const CN_HK: Self = SourceMask(0b11);
}

impl std::ops::BitOr for SourceMask {
    type Output = Self;
    fn bitor(self, rhs: Self) -> Self {
        SourceMask(self.0 | rhs.0)
    }
}

/// Build a 28-byte header.
pub(crate) fn build_header(dict_hash: [u8; 8], source_mask: SourceMask) -> [u8; HEADER_LEN] {
    let mut buf = [0u8; HEADER_LEN];
    buf[0..8].copy_from_slice(&ZHTW_AUTOMATON_MAGIC);
    buf[8..10].copy_from_slice(&CURRENT_HEADER_VERSION.to_le_bytes());
    buf[10..14].copy_from_slice(&DAACHORSE_VERSION_PACKED.to_le_bytes());
    buf[14..22].copy_from_slice(&dict_hash);
    buf[22] = source_mask.0;
    // [23..28] reserved = zero
    buf
}

/// Verify header and return payload slice. Panics on mismatch.
pub(crate) fn verify_header(bytes: &[u8], expected: SourceMask) -> &[u8] {
    if bytes.len() < HEADER_LEN {
        panic!(
            "zhtw: automaton bytes truncated (got {} bytes, need >= {})",
            bytes.len(),
            HEADER_LEN
        );
    }
    let (header, rest) = bytes.split_at(HEADER_LEN);

    if &header[0..8] != &ZHTW_AUTOMATON_MAGIC {
        panic!("zhtw: invalid automaton magic (corrupt binary or wrong format)");
    }

    let header_version = u16::from_le_bytes(header[8..10].try_into().unwrap());
    if header_version != CURRENT_HEADER_VERSION {
        panic!(
            "zhtw: header version mismatch (got {}, expected {})",
            header_version, CURRENT_HEADER_VERSION
        );
    }

    let daachorse_version = u32::from_le_bytes(header[10..14].try_into().unwrap());
    if daachorse_version != DAACHORSE_VERSION_PACKED {
        panic!(
            "zhtw: daachorse version mismatch (built 0x{:08x}, expected 0x{:08x}) — cargo clean && cargo build",
            daachorse_version, DAACHORSE_VERSION_PACKED
        );
    }

    let source_mask = header[22];
    if source_mask != expected.0 {
        panic!(
            "zhtw: automaton source mask mismatch (got {:#04b}, expected {:#04b})",
            source_mask, expected.0
        );
    }

    rest
}

#[cfg(test)]
mod tests {
    use super::*;

    #[test]
    fn round_trip() {
        let hash = [1, 2, 3, 4, 5, 6, 7, 8];
        let header = build_header(hash, SourceMask::CN_HK);
        let payload = b"hello";
        let mut blob = header.to_vec();
        blob.extend_from_slice(payload);

        let rest = verify_header(&blob, SourceMask::CN_HK);
        assert_eq!(rest, payload);
    }

    #[test]
    #[should_panic(expected = "truncated")]
    fn rejects_short_blob() {
        verify_header(&[0; 10], SourceMask::CN_HK);
    }

    #[test]
    #[should_panic(expected = "magic")]
    fn rejects_bad_magic() {
        let mut blob = [0u8; 28];
        blob[0..8].copy_from_slice(b"XXXXXXXX");
        verify_header(&blob, SourceMask::CN_HK);
    }

    #[test]
    #[should_panic(expected = "source mask")]
    fn rejects_wrong_source_mask() {
        let hash = [0; 8];
        let header = build_header(hash, SourceMask::CN);
        verify_header(&header, SourceMask::CN_HK);
    }
}

fn main() {

    println!("cargo:rerun-if-changed=build.rs");

        panic!(
            "sdk/rust/zhtw/data/zhtw-data.json missing ({e}) — run `make export`"
        )
    });
    let raw: RawZhtwData =
        serde_json::from_str(&json_str).expect("zhtw-data.json parse failed");

    let dict_hash = blake3_first_8(&json_str);
    let out_dir = PathBuf::from(env::var("OUT_DIR").unwrap());

    generate_char_map(&raw, &out_dir);
    generate_automaton_cnhk(&raw, dict_hash, &out_dir);

fn blake3_first_8(data: &str) -> [u8; 8] {
    let hash = blake3::hash(data.as_bytes());
    let bytes = hash.as_bytes();
    let mut out = [0u8; 8];
    out.copy_from_slice(&bytes[..8]);
    out
}

fn generate_char_map(raw: &RawZhtwData, out_dir: &PathBuf) {
    let mut file = std::io::BufWriter::new(
        fs::File::create(out_dir.join("generated_maps.rs")).unwrap(),
    );

    use std::io::Write;
    writeln!(file, "// Auto-generated by build.rs — do not edit").unwrap();
    writeln!(file).unwrap();

    let mut builder = phf_codegen::Map::new();
    for (src, tgt) in &raw.char_map {
        let src_char = src.chars().next().expect("char_map key must be single char");
        let tgt_char = tgt.chars().next().expect("char_map value must be single char");
        builder.entry(src_char, &format!("'{}'", tgt_char.escape_unicode()));
    }
    writeln!(
        file,
        "pub(crate) static CHAR_MAP: phf::Map<char, char> = {};",
        builder.build()
    )
    .unwrap();

    writeln!(file).unwrap();
    writeln!(
        file,
        "pub(crate) const DATA_VERSION: &str = {:?};",
        raw.version
    )
    .unwrap();
}

fn generate_automaton_cnhk(raw: &RawZhtwData, dict_hash: [u8; 8], out_dir: &PathBuf) {
    // Collect all terms for cn + hk
    let mut patterns: Vec<(String, String)> = Vec::new();

    for (source_key, terms) in &raw.terms {
        match source_key.as_str() {
            "cn" | "hk" => {
                for t in terms {
                    patterns.push((t.source.clone(), t.target.clone()));
                }
            }
            _ => {} // skip unknown sources
        }
    }

    // Sort for deterministic output
    patterns.sort();

    // Build daachorse automaton with MatchKind::Standard
    use daachorse::{CharwiseDoubleArrayAhoCorasickBuilder, MatchKind};

    let patterns_with_ids: Vec<(&str, u32)> = patterns
        .iter()
        .enumerate()
        .map(|(idx, (src, _))| (src.as_str(), idx as u32))
        .collect();

    let pma = CharwiseDoubleArrayAhoCorasickBuilder::new()
        .match_kind(MatchKind::Standard)
        .build_with_values(patterns_with_ids)
        .expect("daachorse build failed");

    // Serialize automaton
    let automaton_bytes = pma.serialize();

    // Build header
    // Include header module via source
    let header_bytes = build_header_bytes(dict_hash, 0b11); // CN | HK

    // Write automaton blob (header + serialized bytes)
    let mut blob = header_bytes.to_vec();
    blob.extend_from_slice(&automaton_bytes);
    fs::write(out_dir.join("automaton-cnhk.bin"), &blob).unwrap();

    // Write pattern table (bincode-like: length-prefixed strings)
    let pattern_table_bytes = serialize_pattern_table(&patterns);
    fs::write(out_dir.join("pattern-table-cnhk.bin"), &pattern_table_bytes).unwrap();
}

fn build_header_bytes(dict_hash: [u8; 8], source_mask: u8) -> [u8; 28] {
    let mut buf = [0u8; 28];
    buf[0..8].copy_from_slice(b"ZHTWDAAC");
    buf[8..10].copy_from_slice(&1u16.to_le_bytes()); // header version
    buf[10..14].copy_from_slice(&0x0001_0000u32.to_le_bytes()); // daachorse 1.0.0
    buf[14..22].copy_from_slice(&dict_hash);
    buf[22] = source_mask;
    buf
}

fn serialize_pattern_table(patterns: &[(String, String)]) -> Vec<u8> {
    let mut buf = Vec::new();
    // Write count as u32 LE
    buf.extend_from_slice(&(patterns.len() as u32).to_le_bytes());
    for (src, tgt) in patterns {
        // Each entry: u32 src_len + src_bytes + u32 tgt_len + tgt_bytes
        buf.extend_from_slice(&(src.len() as u32).to_le_bytes());
        buf.extend_from_slice(src.as_bytes());
        buf.extend_from_slice(&(tgt.len() as u32).to_le_bytes());
        buf.extend_from_slice(tgt.as_bytes());
    }
    buf
}

//! Daachorse wrapper with identity protection and byte↔codepoint mapping.
//!
//! Ports Python `src/zhtw/matcher.py::find_matches` and TS
//! `sdk/typescript/src/core/matcher.ts::findMatches`.

use std::collections::{BTreeSet, HashMap};

use daachorse::{
    CharwiseDoubleArrayAhoCorasick, CharwiseDoubleArrayAhoCorasickBuilder, MatchKind,
};

use crate::generated;
use crate::header::{self, SourceMask};
use crate::Source;

/// A raw term hit from daachorse, using byte offsets.
#[derive(Debug, Clone)]
pub(crate) struct TermHit {
    pub byte_start: usize,
    pub byte_end: usize,
    pub source: String,
    pub target: String,
}

/// Pattern table entry: (source, target).
pub(crate) type PatternTable = Vec<(String, String)>;

/// Deserialize pattern table from build.rs binary format.
pub(crate) fn deserialize_pattern_table(bytes: &[u8]) -> PatternTable {
    let mut pos = 0;
    let count = u32::from_le_bytes(bytes[pos..pos + 4].try_into().unwrap()) as usize;
    pos += 4;

    let mut table = Vec::with_capacity(count);
    for _ in 0..count {
        let src_len = u32::from_le_bytes(bytes[pos..pos + 4].try_into().unwrap()) as usize;
        pos += 4;
        let src = std::str::from_utf8(&bytes[pos..pos + src_len])
            .expect("pattern table: invalid UTF-8 in source")
            .to_string();
        pos += src_len;

        let tgt_len = u32::from_le_bytes(bytes[pos..pos + 4].try_into().unwrap()) as usize;
        pos += 4;
        let tgt = std::str::from_utf8(&bytes[pos..pos + tgt_len])
            .expect("pattern table: invalid UTF-8 in target")
            .to_string();
        pos += tgt_len;

        table.push((src, tgt));
    }
    table
}

/// Build a daachorse automaton at runtime (for custom sources / custom_dict).
pub(crate) fn build_automaton(
    patterns: &[(String, String)],
) -> CharwiseDoubleArrayAhoCorasick<u32> {
    let patterns_with_ids: Vec<(&str, u32)> = patterns
        .iter()
        .enumerate()
        .map(|(idx, (src, _))| (src.as_str(), idx as u32))
        .collect();

    CharwiseDoubleArrayAhoCorasickBuilder::new()
        .match_kind(MatchKind::Standard)
        .build_with_values(patterns_with_ids)
        .expect("daachorse build failed")
}

/// Deserialize precompiled default automaton from embedded bytes.
pub(crate) fn deserialize_default_automaton()
    -> CharwiseDoubleArrayAhoCorasick<u32>
{
    let payload = header::verify_header(
        generated::AUTOMATON_CNHK_BYTES,
        SourceMask::CN_HK,
    );

    // SAFETY: The bytes were serialized by the same daachorse version
    // (verified by header). The build.rs that produced them ran in this
    // exact compilation, so format and platform match.
    let (pma, trailing) = unsafe {
        CharwiseDoubleArrayAhoCorasick::<u32>::deserialize_unchecked(payload)
    };
    assert!(
        trailing.is_empty(),
        "zhtw: {} trailing bytes after automaton deserialization",
        trailing.len()
    );
    pma
}

/// Find term matches with identity protection.
///
/// This is the core algorithm ported from Python `matcher.py:find_matches`.
/// Uses MatchKind::Standard + find_overlapping_iter to collect ALL matches
/// (including identity blockers), then applies the bisect+prefix-max-end
/// algorithm for protected ranges.
pub(crate) fn find_term_matches(
    pma: &CharwiseDoubleArrayAhoCorasick<u32>,
    pattern_table: &[(String, String)],
    text: &str,
) -> Vec<TermHit> {
    if text.is_empty() {
        return Vec::new();
    }

    // 1. Collect all overlapping hits (daachorse charwise returns byte offsets)
    let mut raw: Vec<TermHit> = Vec::new();
    for m in pma.find_overlapping_iter(text) {
        let (src, tgt) = &pattern_table[m.value() as usize];
        raw.push(TermHit {
            byte_start: m.start(),
            byte_end: m.end(),
            source: src.clone(),
            target: tgt.clone(),
        });
    }

    if raw.is_empty() {
        return Vec::new();
    }

    // 2. Sort by (byte_start ASC, length DESC)
    raw.sort_by(|a, b| {
        a.byte_start
            .cmp(&b.byte_start)
            .then_with(|| {
                let a_len = a.byte_end - a.byte_start;
                let b_len = b.byte_end - b.byte_start;
                b_len.cmp(&a_len)
            })
    });

    // 3. Build protected byte-range set from identity matches
    let mut protected: BTreeSet<usize> = BTreeSet::new();

    let identity_matches: Vec<&TermHit> =
        raw.iter().filter(|m| m.source == m.target).collect();
    let non_identity_spans: Vec<(usize, usize)> = raw
        .iter()
        .filter(|m| m.source != m.target)
        .map(|m| (m.byte_start, m.byte_end))
        .collect();

    if non_identity_spans.is_empty() {
        // All identity ranges are protected
        for id_match in &identity_matches {
            for b in id_match.byte_start..id_match.byte_end {
                protected.insert(b);
            }
        }
    } else {
        let mut ni_sorted = non_identity_spans.clone();
        ni_sorted.sort();
        let ni_starts: Vec<usize> = ni_sorted.iter().map(|(s, _)| *s).collect();

        // prefix_max_end[i] = max(end) among ni_sorted[0..=i]
        let mut prefix_max_end: Vec<usize> = Vec::with_capacity(ni_sorted.len());
        let mut max_e = 0usize;
        for &(_, e) in &ni_sorted {
            max_e = max_e.max(e);
            prefix_max_end.push(max_e);
        }

        for id_match in &identity_matches {
            // Binary search: last index where ni_starts[idx] <= id_match.byte_start
            let idx = match ni_starts.partition_point(|&s| s <= id_match.byte_start) {
                0 => None,
                n => Some(n - 1),
            };

            let is_contained = idx
                .map(|i| prefix_max_end[i] >= id_match.byte_end)
                .unwrap_or(false);

            if !is_contained {
                for b in id_match.byte_start..id_match.byte_end {
                    protected.insert(b);
                }
            }
        }
    }

    // 4. Left-to-right filter
    let mut chosen: Vec<TermHit> = Vec::new();
    let mut last_end: usize = 0;

    for m in &raw {
        if m.byte_start < last_end {
            continue; // overlap with previously chosen
        }
        if m.source != m.target {
            // Non-identity: check protected range
            let touches_protected = (m.byte_start..m.byte_end)
                .any(|b| protected.contains(&b));
            if touches_protected {
                continue; // skip, don't advance cursor
            }
        }
        last_end = m.byte_end; // advance cursor
        if m.source != m.target {
            chosen.push(m.clone()); // identity never emitted
        }
    }

    chosen
}

/// Build byte-to-codepoint mapping for a text string.
/// Returns a Vec where `byte_to_cp[byte_offset]` = codepoint index.
/// Length = text.len() + 1 (to support exclusive end offsets).
pub(crate) fn build_byte_to_cp(text: &str) -> Vec<usize> {
    let mut map = vec![0usize; text.len() + 1];
    for (cp_idx, (byte_idx, ch)) in text.char_indices().enumerate() {
        // Fill all bytes of this char with the same cp index
        for b in byte_idx..byte_idx + ch.len_utf8() {
            map[b] = cp_idx;
        }
        // The byte just past this char maps to cp_idx + 1
        map[byte_idx + ch.len_utf8()] = cp_idx + 1;
    }
    map
}

/// Apply term replacements to produce the after-terms string.
pub(crate) fn apply_term_replacements(text: &str, hits: &[TermHit]) -> String {
    if hits.is_empty() {
        return text.to_string();
    }

    let mut parts: Vec<&str> = Vec::new();
    let mut targets: Vec<&str> = Vec::new();
    let mut last_end = 0;

    for hit in hits {
        parts.push(&text[last_end..hit.byte_start]);
        targets.push(&hit.target);
        last_end = hit.byte_end;
    }
    parts.push(&text[last_end..]);

    let mut result = String::with_capacity(text.len());
    for i in 0..targets.len() {
        result.push_str(parts[i]);
        result.push_str(targets[i]);
    }
    result.push_str(parts[targets.len()]);
    result
}

/// Apply char map to text.
pub(crate) fn apply_charmap(text: &str, char_map: &phf::Map<char, char>) -> String {
    let mut out = String::with_capacity(text.len());
    for ch in text.chars() {
        out.push(*char_map.get(&ch).unwrap_or(&ch));
    }
    out
}

        // the automaton will just have extra patterns that don't hurt
        // (identity protection handles them). This is simpler for v1.
        // TODO: In a future version, tag patterns with source in build.rs.

        let mut patterns = default_table;

        // Merge custom_dict: custom entries override built-in same-key
        if !config.custom_dict.is_empty() {
            let mut dict_map: HashMap<String, String> = patterns.into_iter().collect();
            for (k, v) in config.custom_dict {
                dict_map.insert(k, v);
            }
            patterns = dict_map.into_iter().collect();
            patterns.sort();
        }

        let pma = matcher::build_automaton(&patterns);

        Ok(Self {
            pma: Arc::new(pma),
            pattern_table: Arc::new(patterns),
            char_layer_enabled,
        })
    }

    /// Zero-cost shared default instance (Cn + Hk, no custom dict).
    pub fn default_instance() -> &'static Self {
        &DEFAULT
    }

    pub fn builder() -> crate::Builder {
        crate::Builder::default()
    }

    /// Convert text from Simplified/HK to Taiwan Traditional Chinese.
    pub fn convert(&self, text: &str) -> String {
        if text.is_empty() {
            return String::new();
        }
        let hits = matcher::find_term_matches(&self.pma, &self.pattern_table, text);
        let after_terms = matcher::apply_term_replacements(text, &hits);
        if self.char_layer_enabled {
            matcher::apply_charmap(&after_terms, &generated::CHAR_MAP)
        } else {
            after_terms
        }
    }

    /// Check text and return all matches with codepoint offsets.
    pub fn check(&self, text: &str) -> Vec<Match> {
        if text.is_empty() {
            return Vec::new();
        }

        let byte_to_cp = matcher::build_byte_to_cp(text);

        // Term layer
        let term_hits = matcher::find_term_matches(&self.pma, &self.pattern_table, text);
        let mut matches: Vec<Match> = term_hits
            .iter()
            .map(|h| Match {
                start: byte_to_cp[h.byte_start],
                end: byte_to_cp[h.byte_end],
                source: h.source.clone(),
                target: h.target.clone(),
            })
            .collect();

        // Char layer — walk original text
        if self.char_layer_enabled {
            for (cp_idx, ch) in text.chars().enumerate() {
                if let Some(&mapped) = generated::CHAR_MAP.get(&ch) {
                    if mapped != ch {
                        matches.push(Match {
                            start: cp_idx,
                            end: cp_idx + 1,
                            source: ch.to_string(),
                            target: mapped.to_string(),
                        });
                    }
                }
            }
        }

        matches
    }

    /// Lookup a word and return conversion details.
    pub fn lookup(&self, word: &str) -> LookupResult {
        if word.is_empty() {
            return LookupResult {
                input: String::new(),
                output: String::new(),
                changed: false,
                details: Vec::new(),
            };
        }

        let byte_to_cp = matcher::build_byte_to_cp(word);

        // Term layer hits
        let term_hits = matcher::find_term_matches(&self.pma, &self.pattern_table, word);

        // Build covered byte ranges
        let mut covered_bytes: std::collections::BTreeSet<usize> =
            std::collections::BTreeSet::new();
        for h in &term_hits {
            for b in h.byte_start..h.byte_end {
                covered_bytes.insert(b);
            }
        }

        let mut details: Vec<ConversionDetail> = Vec::new();

        // Term details — target gets char-map post-pass (Python lookup.py:78-83)
        for h in &term_hits {
            let post_target = if self.char_layer_enabled {
                matcher::apply_charmap(&h.target, &generated::CHAR_MAP)
            } else {
                h.target.clone()
            };
            details.push(ConversionDetail {
                source: h.source.clone(),
                target: post_target,
                layer: Layer::Term,
                position: byte_to_cp[h.byte_start],
            });
        }

        // Char layer — skip bytes covered by term hits
        if self.char_layer_enabled {
            for (byte_idx, ch) in word.char_indices() {
                // Skip if any byte of this char is covered by a term hit
                let is_covered = (byte_idx..byte_idx + ch.len_utf8())
                    .any(|b| covered_bytes.contains(&b));
                if is_covered {
                    continue;
                }
                if let Some(&mapped) = generated::CHAR_MAP.get(&ch) {
                    if mapped != ch {
                        details.push(ConversionDetail {
                            source: ch.to_string(),
                            target: mapped.to_string(),
                            layer: Layer::Char,
                            position: byte_to_cp[byte_idx],
                        });
                    }
                }
            }
        }

        // Sort details by position
        details.sort_by_key(|d| d.position);

        // Build output string
        let output = self.convert(word);

        LookupResult {
            input: word.to_string(),
            changed: output != word,
            output,
            details,
        }
    }
}

impl Default for Converter {
    fn default() -> Self {
        Self::default_instance().clone()
    }
}

                        })?;
                        s.parse::<Source>()
                            .map_err(|e| JsValue::from_str(&e.to_string()))
                    })
                    .collect::<Result<Vec<_>, _>>()?;
                builder = builder.sources(sources);
            }
        }

        if let Ok(dict_val) = js_sys::Reflect::get(&opts, &"customDict".into()) {
            if dict_val.is_object() && !dict_val.is_null() && !dict_val.is_undefined() {
                let entries = js_sys::Object::entries(&dict_val.unchecked_into());
                let dict: std::collections::HashMap<String, String> = entries
                    .iter()
                    .filter_map(|entry| {
                        let arr: js_sys::Array = entry.unchecked_into();
                        let k = arr.get(0).as_string()?;
                        let v = arr.get(1).as_string()?;
                        Some((k, v))
                    })
                    .collect();
                builder = builder.custom_dict(dict);
            }
        }
    }

    let inner = builder
        .build()
        .map_err(|e| JsValue::from_str(&format!("zhtw: {e}")))?;
    Ok(Converter { inner })
}
```
